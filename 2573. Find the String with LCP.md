# 💻 ✅ Optimal Java Code (Greedy + DP Validation)

```java
class Solution {
    public String findTheString(int[][] lcp) {
        int n = lcp.length;
        char[] s = new char[n];

        // Step 1: Initialize with '?'
        for (int i = 0; i < n; i++) {
            s[i] = '?';
        }

        char ch = 'a';

        // Step 2: Build string (Greedy)
        for (int i = 0; i < n; i++) {

            // agar already filled hai to skip
            if (s[i] != '?') continue;

            // agar characters khatam ho gaye
            if (ch > 'z') return "";

            // jaha lcp[i][j] > 0 hai, waha same character
            for (int j = i; j < n; j++) {
                if (lcp[i][j] > 0) {
                    s[j] = ch;
                }
            }

            ch++; // next character
        }

        // Step 3: Validate using DP (LCP rebuild)
        int[][] dp = new int[n][n];

        // bottom-up fill
        for (int i = n - 1; i >= 0; i--) {
            for (int j = n - 1; j >= 0; j--) {

                if (s[i] == s[j]) {
                    dp[i][j] = 1;

                    if (i + 1 < n && j + 1 < n) {
                        dp[i][j] += dp[i + 1][j + 1];
                    }
                } else {
                    dp[i][j] = 0;
                }

                // mismatch check
                if (dp[i][j] != lcp[i][j]) {
                    return "";
                }
            }
        }

        return new String(s);
    }
}
```
